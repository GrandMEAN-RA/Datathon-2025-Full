# ---------- BASE IMAGE ----------
FROM python:3.10-slim

# ---------- SYSTEM SETUP ----------
# Install PortAudio and build tools BEFORE installing Python deps
RUN apt-get update && apt-get install -y \
    build-essential \
    portaudio19-dev \
    && rm -rf /var/lib/apt/lists/*

# ---------- APP SETUP ----------
WORKDIR /app
COPY . /app

# Install Python dependencies
RUN pip install --upgrade pip && \
    pip install --no-cache-dir -r requirements.txt

# ---------- RUN ----------
EXPOSE 8501
CMD ["streamlit", "run", "Banking_Iya_Bola_App.py", "--server.port=8501", "--server.address=0.0.0.0"]
